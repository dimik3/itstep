void swap(int* a, int* b);

int main()
{
	int a = 4;
	int b = 5;
	swap(&a, &b);
	printf("a = %i\n",a);
	printf("b = %i\n", b);
}

void swap(int* a, int* b)
{									//Меняем значения a и b местами .
	int buff = *a;
	*a = *b;							
	*b = buff;
}
