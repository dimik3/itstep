#include <stdio.h>
#include <stdlib.h>

// Одномерные статические массивы

#define DAYS_COUNT 7

int main()
{
	float temperature[DAYS_COUNT];

	float summ = 0;

	for (int i = 0; i < DAYS_COUNT; i++)
	{
		printf("Enter please temperature %i day: ",i + 1);
		scanf("%f",&temperature[i]);
		
		summ += temperature[i];
	}
	printf("Average temperature: %.2f", summ / DAYS_COUNT);
}
