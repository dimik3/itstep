//Массивы строк

void strCopy(char* from, char* to)
{
	while (*from != 0 && *from != '\n')
	{
		*to = *from;
		from++;
		to++;
	}
	*to = 0;
}

void clearArray(char array[][256],
	int size)
{
	for (int i = 0; i < size; i++)
	{
		array[i][0] = 0;
	}
}

void printArray(char array[][256],
	int size)
{
	for (int i = 0; i < size; i++)
	{
		printf("nameCell[%i]: %s\n",
			i, array[i]);
	}
}
